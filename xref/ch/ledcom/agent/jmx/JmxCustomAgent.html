211 import java.util.HashMap;

218 /**
219  * This CustomAgent will start an RMI Connector Server using only port
220  * "ch.ledcom.agent.jmx.port". When used with the javaagent option, it allows a
221  * remote connection using JConsole or VisualVM through an SSH tunnel.
222  * <p/>
223  * The purpose of this agent is to open a JMX server that will use only one
224  * fixed port for all communication instead of two dynamically set ports.
225  * <p/>
226  * Code adapted from:
227  * https://blogs.oracle.com/jmxetc/entry/connecting_through_firewall_using_jmx
228  *
229  * @author gehel
230  */
231 public final class JmxCustomAgent {
232 
233     /** Base name for all parameter names. */
234     private static final String BASE = "ch.ledcom.agent.jmx.";
235     /** Parameter name to set the JMX port. */
236     public static final String PORT_KEY = BASE + "port";
237     /** Parameter name to force localhost. */
238     public static final String FORCE_LOCALHOST_KEY = BASE + "forceLocalhost";
239     /** Parameter name to activate authentication. */
240     public static final String AUTHENTICATE = BASE + "authenticate";
241     /** Parameter name to set password file used for authentication. */
242     public static final String PASSWORD_FILE = BASE + "password.file";
243     /** Parameter name to set access file used for authentication. */
244     public static final String ACCESS_FILE = BASE + "access.file";
245 
246     /** Default JMX port. */
247     private static final int DEFAULT_PORT = 3000;
248 
249     /**
250      * Private constructor to make sure nobody tries to instantiate this class.
251      */
252     private JmxCustomAgent() {
253     }
254 
255     /**
256      * Entry point into a JavaAgent.
257      * @param agentArgs arguments given to the agent (not used, we only use
258      *                  system properties)
259      * @throws IOException
260      */
261     public static void premain(final String agentArgs) throws IOException {
262         // get all parameters from environment
263         final int port = parseInt(System.getProperty(PORT_KEY), DEFAULT_PORT);
264         final boolean forceLocalhost = Boolean.getBoolean(FORCE_LOCALHOST_KEY);
265         final String accessFile = System.getProperty(ACCESS_FILE);
266         final String passwordFile = System.getProperty(PASSWORD_FILE);
267         final boolean authenticate = Boolean.getBoolean(AUTHENTICATE);
268 
269         System.out.println("RMI / JMX Custom Agent");
270         System.out.println("======================");
271         System.out.println(PORT_KEY + " = [" + port + "]");
272         System.out.println(FORCE_LOCALHOST_KEY + " = [" + forceLocalhost + "]");
273         System.out.println(AUTHENTICATE + " = [" + authenticate + "]");
274         System.out.println(ACCESS_FILE + " = [" + accessFile + "]");
275         System.out.println(PASSWORD_FILE + " = [" + passwordFile + "]");
276 
277         // Ensure cryptographically strong random number generator used
278         // to choose the object number - see java.rmi.server.ObjID
279         System.setProperty("java.rmi.server.randomIDs", "true");
280 
281         System.out.println("Create RMI registry on port " + port);
282         LocateRegistry.createRegistry(port);
283 
284         // Retrieve the PlatformMBeanServer.
285         MBeanServer mbs = ManagementFactory.getPlatformMBeanServer();
286 
287         // Environment map.
288         System.out.println("Initialize the environment map");
289         HashMap<String, Object> env = new HashMap<String, Object>();
290 
291         // Provide the password file used by the connector server to
292         // perform user authentication. The password file is a properties
293         // based text file specifying username/password pairs.
294         env.put("jmx.remote.x.password.file", passwordFile);
295 
296         // Provide the access level file used by the connector server to
297         // perform user authorization. The access level file is a properties
298         // based text file specifying username/access level pairs where
299         // access level is either "readonly" or "readwrite" access to the
300         // MBeanServer operations.
301         env.put("jmx.remote.x.access.file", accessFile);
302         env.put("jmx.remote.x.authenticate", authenticate);
303 
304         // Create an RMI connector server.
305         //
306         // As specified in the JMXServiceURL the RMIServer stub will be
307         // registered in the RMI registry running in the local host on
308         // port 3000 with the name "jmxrmi". This is the same name the
309         // out-of-the-box management agent uses to register the RMIServer
310         // stub too.
311         //
312         // The port specified in "service:jmx:rmi://"+hostname+":"+port
313         // is the second port, where RMI connection objects will be exported.
314         // Here we use the same port as that we choose for the RMI registry.
315         // The port for the RMI registry is specified in the second part
316         // of the URL, in "rmi://"+hostname+":"+port
317         String hostname;
318         if (forceLocalhost) {
319             System.out
320                     .println("Using localhost as hostname as defined by system property: "
321                             + FORCE_LOCALHOST_KEY);
322             hostname = "localhost";
323         } else {
324             hostname = InetAddress.getLocalHost().getHostName();
325         }
326 
327         System.out.println("Create an RMI/JMX connector server, hostname is: ["
328                 + hostname + "]");
329         String serviceUrl = "service:jmx:rmi://" + hostname + ":" + port
330                 + "/jndi/rmi://" + hostname + ":" + port + "/jmxrmi";
331         System.out.println("and url is: [" + serviceUrl + "]");
332         JMXServiceURL url = new JMXServiceURL(serviceUrl);
333 
334         // Now create the server from the JMXServiceURL
335         JMXConnectorServer cs = JMXConnectorServerFactory
336                 .newJMXConnectorServer(url, env, mbs);
337 
338         // Start the RMI connector server.
339         System.out.println("Start the RMI/JMX connector server on port ["
340                 + port + "]");
341         cs.start();
342     }
343 
344     /**
345      * Utility method to parse integer.
346      * @param s String to parse as an int
347      * @param defaultValue default value to return in case the String is empty
348      * @return the integer represented by the String or the default value if the String is empty
349      */
350     private static int parseInt(final String s, final int defaultValue) {
351         if (s == null || s.trim().equals("")) {
352             return defaultValue;
353         }
354         return Integer.parseInt(s);
355     }
