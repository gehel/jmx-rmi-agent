16  package ch.ledcom.agent.jmx;
17  
18  import javax.management.MBeanServer;
19  import javax.management.remote.JMXConnectorServer;
20  import javax.management.remote.JMXConnectorServerFactory;
21  import javax.management.remote.JMXServiceURL;
22  import java.io.IOException;
23  import java.lang.management.ManagementFactory;
24  import java.net.InetAddress;
25  import java.rmi.registry.LocateRegistry;
26  import java.util.HashMap;
27  
28  /**
29   * This CustomAgent will start an RMI Connector Server using only port
30   * "ch.ledcom.agent.jmx.port". When used with the javaagent option, it allows a
31   * remote connection using JConsole or VisualVM through an SSH tunnel.
32   * <p/>
33   * The purpose of this agent is to open a JMX server that will use only one
34   * fixed port for all communication instead of two dynamically set ports.
35   * <p/>
36   * Code adapted from:
37   * https://blogs.oracle.com/jmxetc/entry/connecting_through_firewall_using_jmx
38   *
39   * @author gehel
40   */
41  public final class JmxCustomAgent {
42  
43      /**
44       * Base name for all parameter names.
45       */
46      private static final String BASE = "ch.ledcom.agent.jmx.";
47  
48      /** Parameter name to set the JMX port. */
49      public static final String PORT_KEY = BASE + "port";
50  
51      /** Parameter name to force localhost. */
52      public static final String FORCE_LOCALHOST_KEY = BASE + "forceLocalhost";
53  
54      /** Parameter name to activate authentication. */
55      public static final String AUTHENTICATE = BASE + "authenticate";
56  
57      /** Parameter name to set password file used for authentication. */
58      public static final String PASSWORD_FILE = BASE + "password.file";
59  
60      /** Parameter name to set access file used for authentication. */
61      public static final String ACCESS_FILE = BASE + "access.file";
62  
63      /** Default JMX port. */
64      private static final int DEFAULT_PORT = 3000;
65  
66      /**
67       * Private constructor to make sure nobody tries to instantiate this class.
68       */
69      private JmxCustomAgent() {
70      }
71  
72      /**
73       * Entry point into a JavaAgent.
74       *
75       * @param agentArgs arguments given to the agent (not used, we only use
76       *                  system properties)
77       * @throws IOException in case of problems setting up the agent
78       */
79      public static void premain(final String agentArgs) throws IOException {
80          // get all parameters from environment
81          final int port = parseInt(System.getProperty(PORT_KEY), DEFAULT_PORT);
82          final boolean forceLocalhost = Boolean.getBoolean(FORCE_LOCALHOST_KEY);
83          final String accessFile = System.getProperty(ACCESS_FILE);
84          final String passwordFile = System.getProperty(PASSWORD_FILE);
85          final boolean authenticate = Boolean.getBoolean(AUTHENTICATE);
86  
87          System.out.println("RMI / JMX Custom Agent");
88          System.out.println("======================");
89          System.out.println(PORT_KEY + " = [" + port + "]");
90          System.out.println(FORCE_LOCALHOST_KEY + " = [" + forceLocalhost + "]");
91          System.out.println(AUTHENTICATE + " = [" + authenticate + "]");
92          System.out.println(ACCESS_FILE + " = [" + accessFile + "]");
93          System.out.println(PASSWORD_FILE + " = [" + passwordFile + "]");
94  
95          // Ensure cryptographically strong random number generator used
96          // to choose the object number - see java.rmi.server.ObjID
97          System.setProperty("java.rmi.server.randomIDs", "true");
98  
99          System.out.println("Create RMI registry on port " + port);
100         LocateRegistry.createRegistry(port);
101 
102         // Retrieve the PlatformMBeanServer.
103         MBeanServer mbs = ManagementFactory.getPlatformMBeanServer();
104 
105         // Environment map.
106         System.out.println("Initialize the environment map");
107         HashMap<String, Object> env = new HashMap<String, Object>();
108 
109         // Provide the password file used by the connector server to
110         // perform user authentication. The password file is a properties
111         // based text file specifying username/password pairs.
112         env.put("jmx.remote.x.password.file", passwordFile);
113 
114         // Provide the access level file used by the connector server to
115         // perform user authorization. The access level file is a properties
116         // based text file specifying username/access level pairs where
117         // access level is either "readonly" or "readwrite" access to the
118         // MBeanServer operations.
119         env.put("jmx.remote.x.access.file", accessFile);
120         env.put("jmx.remote.x.authenticate", authenticate);
121 
122         // Create an RMI connector server.
123         //
124         // As specified in the JMXServiceURL the RMIServer stub will be
125         // registered in the RMI registry running in the local host on
126         // port 3000 with the name "jmxrmi". This is the same name the
127         // out-of-the-box management agent uses to register the RMIServer
128         // stub too.
129         //
130         // The port specified in "service:jmx:rmi://"+hostname+":"+port
131         // is the second port, where RMI connection objects will be exported.
132         // Here we use the same port as that we choose for the RMI registry.
133         // The port for the RMI registry is specified in the second part
134         // of the URL, in "rmi://"+hostname+":"+port
135         String hostname;
136         if (forceLocalhost) {
137             System.out
138                     .println("Using localhost as hostname "
139                             + "as defined by system property: "
140                             + FORCE_LOCALHOST_KEY);
141             hostname = "localhost";
142         } else {
143             hostname = InetAddress.getLocalHost().getHostName();
144         }
145 
146         System.out.println("Create an RMI/JMX connector server, hostname is: ["
147                 + hostname + "]");
148         String serviceUrl = "service:jmx:rmi://" + hostname + ":" + port
149                 + "/jndi/rmi://" + hostname + ":" + port + "/jmxrmi";
150         System.out.println("and url is: [" + serviceUrl + "]");
151         JMXServiceURL url = new JMXServiceURL(serviceUrl);
152 
153         // Now create the server from the JMXServiceURL
154         JMXConnectorServer cs = JMXConnectorServerFactory
155                 .newJMXConnectorServer(url, env, mbs);
156 
157         // Start the RMI connector server.
158         System.out.println("Start the RMI/JMX connector server on port ["
159                 + port + "]");
160         cs.start();
161     }
162 
163     /**
164      * Utility method to parse integer.
165      *
166      * @param s            String to parse as an int
167      * @param defaultValue default value to return in case the String is empty
168      * @return the integer represented by the String or the default value if
169      * the String is empty
170      */
171     private static int parseInt(final String s, final int defaultValue) {
172         if (s == null || s.trim().equals("")) {
173             return defaultValue;
174         }
175         return Integer.parseInt(s);
176     }
177 }
